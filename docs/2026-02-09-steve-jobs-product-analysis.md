## 二、產品定位：重新定義類別

### ❌ 錯誤的定位（容易被理解）：
```
Spark = 聯絡人管理 + AI 洞察 + AI 文章分享
```

**賈伯斯會說：**
> **「你們這不是創造，而是貼補丁。買家看到的是三個功能拼湊，他們買的是一個承諾、一個體驗、一個全新生活理念。」**

```
┌─────────────────────────────────────────────────────────────┐
│                                                             │

│                                                             │
│             不是「工具」而是「顧問」                         │
│         不是「管理」而是「發現與培育」                        │
```

```typescript
// types.ts

// ========== 類別一：價值知識庫 ==========
interface ValueDomain {
  id: string;

  usageCount: number;        // 被使用幾次？
}

interface Story {
  id: string;
  domainId: string;
  title: string;             // 「迷失小路學到的事」
  experience: string;        // 真實經驗描述
  lessons: string[];         // 學到的教訓
  shareableContent: {
    platform: 'linkedin' | 'facebook' | 'line';
    content: string;
  }[];
  tags: string[];
  isPublic: boolean;         // 是否公開給聯絡人看到
  createdAt: string;
  usedCount: number;         // 被使用幾次？
}

interface Insight {
  id: string;
  storyId: string;
  content: string;           // 從故事中提煉出的「要點」
  category: string;          // 哲學、實踐、等...
}

// ========== 類別二：聯絡人 + AI 洞察 ==========
interface BusinessContact {

  intelligence: {
    linkedin?: LinkedInActivity;
    facebook?: FacebookActivity;
    interests: string[];
    recentPosts: SocialPost[];
    lastSyncAt: string;
  };

  // AI 分析「關係深度」
  relationship: {
    score: number;           // 0-100
    stage: 'stranger' | 'acquaintance' | 'friend' | 'advocate';
    lastInteraction: string;
    interactionFrequency: number;
    responseRate: number;
    mutualTopics: string[];
  };

  // 轉介紹潛力
  referral: {
    potential: number;       // 0-100
    successfulReferrals: number;
    potentialConnections: string[];  // 他可能介紹給誰
  };

  // 互動歷史
  interactions: Interaction[];
}

interface AIMatchResult {

    story: Story;
    relevanceScore: number;  // 0-100
    why: string;             // 為什麼匹配？
    suggestedAction: {
      type: 'comment' | 'message' | 'post';
      platform: string;
      content: string;
      expectedOutcome: string;
    };
  }[];
  priority: 'urgent' | 'high' | 'medium' | 'low';
}

  // 影響力指標
  metrics: {

    networkViews: number;    // 二度網絡見到人數
    newConnections: string[]; // 獲得的新連結
    engagementRate: number;
  };

  createdAt: string;
}
```
